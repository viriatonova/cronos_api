SECRET_KEY=<secretkey>
DB_SERVER=mysql+pymysql
DB_USER=root
DB_PASSWORD=secret
DB_ROOT_PASSWORD=secret
DB_DATABASE=cronos
DB_HOST=db
DB_PORT=3306
